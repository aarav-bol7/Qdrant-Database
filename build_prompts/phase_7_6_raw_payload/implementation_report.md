# Test 1 (in-process upload via UploadPipeline.execute → assert raw_payload persisted)
body = {"source_type": "pdf", "source_filename": "x.pdf", "content_hash": "sha256:rp1",
        "items": [{"content": "raw_payload test 1 unique content"}]}
UploadPipeline.execute(tenant_id="test_t", bot_id="test_b", doc_id=d, body=body)
doc = Document.objects.get(doc_id=d)
assert doc.raw_payload == body                      # PASS

# Test 2 (re-upload same → no_change → raw_payload unchanged)
r2 = UploadPipeline.execute(... same body ...)
assert r2.status == "no_change"                     # PASS
assert Document.objects.get(doc_id=d).raw_payload == body   # PASS

# Test 3 (re-upload same doc_id, different content → replaced → raw_payload reflects v2)
body2["items"][0]["content"] = "raw_payload test 3 v2 different"
r3 = UploadPipeline.execute(... body2 ...)
assert r3.status == "replaced"                      # PASS
assert doc.raw_payload["items"][0]["content"] == "raw_payload test 3 v2 different"   # PASS

# Admin rendering (3 cases)
